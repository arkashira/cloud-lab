- **Subnet Count & Placement**: Exactly four subnets are provisioned:

  - 2 private subnets, one in each of the same two AZs.  
- **Subnet CIDR Allocation**: Public subnets use the first two /24 blocks of the VPC CIDR; private subnets use the next two /24 blocks.  
- **Outputs**: The module must expose the following outputs:  
  - `vpc_id` – ID of the created VPC.  
  - `public_subnet_ids` – list of public subnet IDs.  
  - `private_subnet_ids` – list of private subnet IDs.  
  - `public_route_table_ids` – list of route table IDs attached to public subnets.  
  - `private_route_table_ids` – list of route table IDs attached to private subnets.  
- **Terraform Validation**: Running `terraform validate` on the module directory must succeed without errors.  
- **Terraform Plan**: Running `terraform plan` with a dummy provider configuration must produce a plan that creates the resources without any errors or warnings.

import pytest
from terraform import TerraformModule  # Hypothetical helper

@pytest.fixture
def vpc_module():

def test_vpc_cidr(vpc_module):
    plan = vpc_module.plan({"cidr_block": "10.0.0.0/16"})
    vpc = plan.resource_changes("aws_vpc", "main")
    assert vpc["change"]["actions"] == ["create"]
    assert vpc["change"]["after"]["cidr_block"] == "10.0.0.0/16"

def test_subnet_count(vpc_module):
    plan = vpc_module.plan({"cidr_block": "10.0.0.0/16"})
    subnets = plan.resource_changes("aws_subnet")
    assert len(subnets) == 4

def test_subnet_azs(vpc_module):
    plan = vpc_module.plan({"cidr_block": "10.0.0.0/16"})
    azs = {s["change"]["after"]["availability_zone"] for s in plan.resource_changes("aws_subnet")}
    assert len(azs) == 2

def test_output_names(vpc_module):
    outputs = vpc_module.outputs()
    expected = {"vpc_id", "public_subnet_ids", "private_subnet_ids",
                "public_route_table_ids", "private_route_table_ids"}
    assert expected == set(outputs.keys())
