# /opt/axentx/cloud-lab/ansible/playbooks/deploy_sample_app.yml
---
- name: Deploy Sample Web Application
  hosts: all
  become: yes
  gather_facts: yes
  vars:
    docker_image: "nginx:alpine"
    container_name: "sample-web-app"
    port: "8080"
    host_port: "8080"
    exposed_port: "80"

  tasks:
    - name: Ensure Docker is installed
      ansible.builtin.import_role:
        name: docker_install
      tags:
        - docker
        - install

    - name: Pull Docker image
      community.docker.docker_image:
        name: "{{ docker_image }}"
        source: pull
        pull_policy: always
      register: pull_result
      ignore_errors: yes
      tags:
        - docker
        - pull

    - name: Remove existing container if present
      community.docker.docker_container:
        name: "{{ container_name }}"
        state: absent
      tags:
        - docker
        - cleanup

    - name: Create and start sample web application container
      community.docker.docker_container:
        name: "{{ container_name }}"
        image: "{{ docker_image }}"
        state: started
        ports:
          - "{{ host_port }}:{{ exposed_port }}"
        restart_policy: always
        publish_all_ports: no
      register: container_result
      tags:
        - docker
        - deploy

    - name: Verify container is running
      ansible.builtin.wait_for:
        port: "{{ host_port }}"
        timeout: 30
        delay: 5
      when: container_result is defined
      tags:
        - docker
        - verify

    - name: Display container status
      ansible.builtin.debug:
        msg: "Sample web application is running on port {{ host_port }}"
      when: container_result is defined
      tags:
        - docker
        - info

# /opt/axentx/cloud-lab/ansible/playbooks/install_docker.yml
---
- name: Install Docker CE on Amazon Linux 2
  hosts: all
  become: yes
  gather_facts: yes
  vars:
    docker_version: "24.0.7"

  tasks:
    - name: Install prerequisites
      ansible.builtin.apt:
        name:
          - apt-transport-https
          - ca-certificates
          - curl
          - gnupg
          - lsb-release
        state: present
      when: ansible_os_family == "Debian"
      tags:
        - docker
        - prerequisites

    - name: Install prerequisites for Amazon Linux
      ansible.builtin.yum:
        name:
          - yum-utils
          - device-mapper-persistent-data
          - lvm2
        state: present
      when: ansible_os_family == "RedHat"
      tags:
        - docker
        - prerequisites

    - name: Add Docker GPG key
      ansible.builtin.get_url:
        url: "https://download.docker.com/linux/amazon/gpg"
        dest: "/etc/pki/rpm-gpg/RPM-GPG-KEY-docker"
        mode: "0644"
      when: ansible_os_family == "RedHat"
      tags:
        - docker
        - key

    - name: Add Docker repository
      ansible.builtin.yum_repository:
        name: docker-ce
        description: Docker CE repository
        baseurl: "https://download.docker.com/linux/amazon/{{ ansible_distribution_major_version }}/stable/x86_64"
        gpgcheck: yes
        gpgkey: "file:///etc/pki/rpm-gpg/RPM-GPG-KEY-docker"
        enabled: yes
      when: ansible_os_family == "RedHat"
      tags:
        - docker
        - repo

    - name: Install Docker CE
      ansible.builtin.yum:
        name:
          - docker-ce
          - docker-ce-cli
          - containerd.io
        state: present
      when: ansible_os_family == "RedHat"
      tags:
        - docker
        - install

    - name: Enable and start Docker service
      ansible.builtin.systemd:
        name: docker
        enabled: yes
        state: started
      when: ansible_os_family == "RedHat"
      tags:
        - docker
        - service

    - name: Verify Docker installation
      ansible.builtin.command: docker --version
      register: docker_version_check
      changed_when: false
      tags:
        - docker
        - verify
```
