const { launchWizard, writeConfig, CONFIG_PATH } = require('../../src/cli/launch');
const fs = require('fs');
const yaml = require('js-yaml');
const path = require('path');

describe('writeConfig', () => {
  it('serializes a dict to YAML and writes to file', () => {
    const sample = { region: 'us-west-2', instance: 't2.micro' };
    writeConfig(sample);
    const content = fs.readFileSync(CONFIG_PATH, 'utf8');
    const loaded = yaml.safeLoad(content);
    expect(loaded).toEqual(sample);
  });

  it('creates file with default permissions', () => {
    writeConfig({});
    const stats = fs.statSync(CONFIG_PATH);
    // 0o600 = owner read/write, no group/world
    expect((stats.mode & 0o777)).toBe(0o600);
  });
});

describe('launchWizard', () => {
  beforeEach(() => {
    // stub prompts to return predetermined values
    jest.spyOn(launchWizard, 'prompt').mockImplementation((msg) => {
      if (msg.includes('region')) return 'us-east-1';
      if (msg.includes('eks')) return true;
      return 't2.micro';
    });
    // ensure no real file writes
    jest.spyOn(fs, 'writeFileSync').mockImplementation(() => {});
  });

  it('throws ClickException for unsupported region + EKS', async () => {
    await expect(launchWizard()).rejects.toThrow('ClickException');
    await expect(launchWizard()).rejects.toThrow('Region us-east-1 does not support EKS');
  });

  it('writes config.yaml when all prompts succeed', async () => {
    await launchWizard();

      CONFIG_PATH,
      expect.any(String),
      { encoding: 'utf8' }
    );
  });
});
