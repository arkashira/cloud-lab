// tests/vpcTemplateSecurityGroup.test.js
const fs = require('fs');
const path = require('path');
const { describe, it, expect } = require('@jest/globals');

const template = JSON.parse(
  fs.readFileSync(path.resolve(__dirname, '../vpc_template.json'), 'utf8')
);

describe('VPC Template Security Group', () => {
  const sg = template.Resources.CloudLabSg;

  it('should exist and be of type AWS::EC2::SecurityGroup', () => {
    expect(sg).toBeDefined();
    expect(sg.Type).toBe('AWS::EC2::SecurityGroup');
  });

  it('should have the correct group name', () => {
    expect(sg.Properties.GroupName).toBe('cloudlab-ssh-http-sg');
  });

  it('should allow inbound SSH from 0.0.0.0/0', () => {
    const sshRule = sg.Properties.SecurityGroupIngress.find(
      r => r.FromPort === 22 && r.ToPort === 22 && r.CidrIp === '0.0.0.0/0'
    );
    expect(sshRule).toBeDefined();
  });

  it('should allow inbound HTTP from 0.0.0.0/0', () => {
    const httpRule = sg.Properties.SecurityGroupIngress.find(
      r => r.FromPort === 80 && r.ToPort === 80 && r.CidrIp === '0.0.0.0/0'
    );
    expect(httpRule).toBeDefined();
  });

  it('should not contain any other inbound rules', () => {
    const ingress = sg.Properties.SecurityGroupIngress;
    expect(ingress).toHaveLength(2);
  });

  it('should have default outbound rule allowing all traffic', () => {
    const egress = sg.Properties.SecurityGroupEgress;
    expect(egress).toBeUndefined(); // AWS defaults to allow all if omitted
  });
});
