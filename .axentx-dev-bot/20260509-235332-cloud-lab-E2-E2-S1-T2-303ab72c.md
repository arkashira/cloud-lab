import React, { createContext, useContext, useState, useEffect } from 'react';
import { User, getCurrentUser } from '../services/authService';

interface AuthContextType {
  user: User | null;
  loading: boolean;
}

const AuthContext = createContext<AuthContextType>({
  user: null,
  loading: true,
});

export const AuthProvider: React.FC<{ children: React.ReactNode }> = ({ children }) => {
  const [user, setUser] = useState<User | null>(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    const fetchUser = async () => {
      const currentUser = await getCurrentUser();
      setUser(currentUser);
      setLoading(false);
    };

    fetchUser();
  }, []);

  return (
    <AuthContext.Provider value={{ user, loading }}>
      {children}
    </AuthContext.Provider>
  );
};

export const useAuth = () => useContext(AuthContext);

## 1. Acceptance Criteria

2. **Share via Unique Link** – When a user clicks “Generate Link”, a unique, non‑guessable URL is returned and copied to clipboard; the link expires after 48 h.  
3. **Dashboard Visibility** – A shared environment appears in the recipient’s dashboard list within 5 seconds of acceptance (or link click) and is marked with a “Shared” badge.  
4. **Access Control** – Only users listed as collaborators (owner + shared users) can open the environment editor; attempts by others result in a 403 response and an “Access denied” UI message.  
5. **Modification Persistence** – Changes made by any authorized collaborator are saved via the API and instantly reflected for all collaborators (real‑time via WebSocket or polling).  
6. **Revocation** – The owner can revoke a share; the environment disappears from the revoked user’s dashboard within 5 seconds.  
7. **Error Handling** – Invalid email, expired link, or network failure shows a descriptive error toast; the UI remains responsive.

    );

  it('generateShareLink returns a URL and expiry', async () => {
    const mockLink = 'https://cloud-lab.com/share/abc123';

    const result = await generateShareLink(envId, token);
    expect(result.link).toBe(mockLink);
    expect(result.expiresAt).toBeGreaterThan(Date.now());
  });
