- **AC1**: A collaborator can be assigned one of three permission levels – **read**, **write**, or **admin** – via the environment API.
- **AC2**: The stored permission level is persisted in the `Environment` document and returned in the collaborator list endpoint.
- **AC3**: When a user with **read** permission accesses an environment, any write‑only actions (e.g., creating resources, updating configs) return HTTP 403.
- **AC4**: When a user with **write** permission accesses an environment, they can perform read and write actions but cannot manage collaborators (admin‑only actions) – attempts return HTTP 403.
- **AC5**: When a user with **admin** permission accesses an environment, they can read, write, and manage collaborators (add/remove/change permissions).
- **AC6**: Permission changes made via the API are propagated to all active sessions within **5 seconds** (real‑time reflection).
- **AC7**: Audit log entry is created for every permission change, containing actor ID, target collaborator ID, old level, new level, and timestamp.

const mongoose = require('mongoose');
const { Environment } = require('./models');

describe('Environment Model - Permission Schema', () => {
  it('should allow only read, write, admin as enum values', async () => {
    const env = new Environment({
      name: 'test-env',
      collaborators: [{ userId: new mongoose.Types.ObjectId(), role: 'read' }],
    });
    await expect(env.validate()).resolves.toBeUndefined();

    env.collaborators[0].role = 'superuser';
    await expect(env.validate()).rejects.toThrow(/`role` is invalid/);
  });

  it('should default new collaborator role to read if omitted', async () => {
    const env = new Environment({
      name: 'default-role',
      collaborators: [{ userId: new mongoose.Types.ObjectId() }],
    });
    await env.save();
    expect(env.collaborators[0].role).toBe('read');
  });

  it('should correctly update a collaborator role', async () => {
    const userId = new mongoose.Types.ObjectId();
    const env = await Environment.create({
      name: 'update-role',
      collaborators: [{ userId, role: 'read' }],
    });

    env.setCollaboratorRole(userId, 'admin');
    await env.save();

    const updated = await Environment.findById(env._id);
    expect(updated.collaborators.id(userId).role).toBe('admin');
  });
});
