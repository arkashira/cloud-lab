const EventEmitter = require('events');

class ParticipantsManager extends EventEmitter {
  constructor() {
    super();
    this.labs = {}; // { [labId]: { [userId]: { id, name, editingSection } } }
    this.io = null;
  }

  /**
   * Attach the Socket.IO server instance.
   * Must be called during server initialization.
   * @param {import('socket.io').Server} socketIO
   */
  setIO(socketIO) {
    this.io = socketIO;
  }

  /**
   * Ensures a lab exists in internal storage.
   * @private
   */
  _ensureLab(labId) {
    if (!this.labs[labId]) {
      this.labs[labId] = {};
    }
  }

  /**
   * Add or update a participant in a lab.
   * @param {string} labId
   * @param {string} userId
   * @param {{ id: string, name: string }} userData
   */
  addParticipant(labId, userId, userData) {
    this._ensureLab(labId);
    const lab = this.labs[labId];
    lab[userId] = {
      ...lab[userId],
      ...userData,
      editingSection: lab[userId]?.editingSection || null,
    };

    this._broadcastUpdate(labId);
    this.emit('updated', labId, this.getParticipants(labId));
  }

  /**
   * Remove a participant from a lab.
   * @param {string} labId
   * @param {string} userId
   */
  removeParticipant(labId, userId) {
    const lab = this.labs[labId];
    if (lab && lab[userId]) {
      delete lab[userId];
      this._broadcastUpdate(labId);
      this.emit('updated', labId, this.getParticipants(labId));
    }
  }

  /**
   * Set the section a user is currently editing.
   * @param {string} labId
   * @param {string} userId
   * @param {string|null} section
   */
  setEditingSection(labId, userId, section) {
    const lab = this.labs[labId];
    if (lab && lab[userId]) {
      lab[userId].editingSection = section;
      this._broadcastUpdate(labId);

    const lab = this.labs[labId] || {};
