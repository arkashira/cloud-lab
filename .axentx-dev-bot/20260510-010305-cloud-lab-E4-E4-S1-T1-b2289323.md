## Final docker-compose.yml

```yaml
# /opt/axentx/cloud-lab/docker/gitlab-compose.yml
version: '3.8'

services:
  gitlab:
    image: gitlab/gitlab-ce:latest
    container_name: gitlab-sandbox
    restart: always
    hostname: '${GITLAB_HOSTNAME:-gitlab.local}'
    # Add shm_size for Sidekiq (critical for CI/CD)
    shm_size: '256m'
    # Add resource limits for stability
    deploy:
      resources:
        limits:
          memory: 4G
        reservations:
          memory: 2G
    environment:
      GITLAB_OMNIBUS_CONFIG: |
        external_url "http://${GITLAB_HOSTNAME:-gitlab.local}:${GITLAB_PORT:-80}"
        gitlab_rails['gitlab_shell_ssh_port'] = ${GITLAB_SSH_PORT:-22}
        gitlab_rails['time_zone'] = 'UTC'
        gitlab_rails['backup_keep_time'] = 172800
        # Artifact retention policy (48 hours)
        gitlab_rails['artifacts_expiration_policy'] = { 'keep_n': 30, 'older_than': '48h', 'enabled': true }
        nginx['enable'] = true
        nginx['client_max_body_size'] = '256m'
        logging['level'] = 'info'
        # Disable HTTPS if not configured (fixes port 443 binding issue)
        nginx['redirect_http_to_https'] = false
    ports:
      - "${GITLAB_PORT:-80}:80"
      - "${GITLAB_SSH_PORT:-22}:22"
      # Commented out - enable only if HTTPS is properly configured
      # - "${GITLAB_HTTPS_PORT:-443}:443"
    volumes:
      - './gitlab/config:/etc/gitlab'
      - './gitlab/logs:/var/log/gitlab'
      - './gitlab/data:/var/opt/gitlab'
    networks:
      - gitlab-network
    healthcheck:
      # Use wget instead of curl (more reliably available)
      test: ["CMD", "wget", "--quiet", "--tries=1", "--spider", "http://localhost"]
      interval: 30s
      timeout: 10s
      retries: 5
      start_period: 120s  # GitLab takes time to start

networks:
  gitlab-network:
    driver: bridge
```
