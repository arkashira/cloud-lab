type ResetStateHandler struct {

	SampleDir string

		SampleDir: sampleDir,
	}
}

// HandleResetState executes the state reset operation
func (h *ResetStateHandler) HandleResetState(w http.ResponseWriter, r *http.Request) {
	ctx := r.Context()
	
	log.Println("Starting state reset operation...")
	
	// Validate sample state files exist
	if err := h.validateSampleFiles(); err != nil {
		log.Printf("Validation failed: %v", err)
		http.Error(w, fmt.Sprintf("Validation error: %v", err), http.StatusBadRequest)
		return
	}

	// Remove existing state files
	if err := h.clearExistingState(); err != nil {
		log.Printf("Failed to clear state: %v", err)
		http.Error(w, fmt.Sprintf("Failed to clear state: %v", err), http.StatusInternalServerError)
		return
	}

	// Copy sample state files to workspace
	if err := h.copySampleState(ctx); err != nil {
		log.Printf("Failed to copy state files: %v", err)
		http.Error(w, fmt.Sprintf("Failed to copy state files: %v", err), http.StatusInternalServerError)
		return
	}

	log.Println("State reset completed successfully")
	w.WriteHeader(http.StatusOK)
	fmt.Fprintf(w, "State files reset successfully")
}

func (h *ResetStateHandler) validateSampleFiles() error {
	entries, err := os.ReadDir(h.SampleDir)
	if err != nil {
		return fmt.Errorf("failed to read sample state directory: %w", err)

	if len(entries) == 0 {
		return fmt.Errorf("no sample state files found in %s", h.SampleDir)
