FROM ubuntu:22.04

# Prevent interactive prompts during package installation
ENV DEBIAN_FRONTEND=noninteractive

# Install dependencies
RUN apt update && apt install -y \
    ansible \
    python3 \
    python3-pip \
    curl \
    git \
    openssh-client \
    && rm -rf /var/lib/apt/lists/*

# Create non-root user for security
RUN useradd -m -s /bin/bash ansible && \
    echo "ansible ALL=(ALL) NOPASSWD:ALL" >> /etc/sudoers

# Create working directories
RUN mkdir -p /opt/axentx/cloud-lab/playbooks /home/ansible/.ssh

# Set ownership
RUN chown -R ansible:ansible /opt/axentx/cloud-lab/playbooks /home/ansible

# Switch to non-root user
USER ansible
WORKDIR /home/ansible

# Copy local playbooks (correct method - use RUN curl for remote files)
# NOTE: For actual deployment, mount playbooks as volumes or use build context
COPY playbooks/ /opt/axentx/cloud-lab/playbooks/

# Create inventory file
RUN echo "[all]\nlocalhost ansible_connection=local" > /opt/axentx/cloud-lab/playbooks/hosts

# Expose Ansible port (default 22 for SSH)
EXPOSE 22

# Default command - run ansible in interactive mode
CMD ["/bin/bash"]
```
