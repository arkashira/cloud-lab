// /opt/axentx/cloud-lab/src/workspaces/quota-ui.js
import React, { useState, useEffect } from 'react';
import { useParams } from 'react-router-dom';
import axios from 'axios';
import { Card, Form, Button, Alert } from 'react-bootstrap';

const QuotaUI = () => {
  const { workspaceId } = useParams();
  const [quota, setQuota] = useState({ cpu: '', ram: '', storage: '' });
  const [error, setError] = useState(null);
  const [success, setSuccess] = useState(false);

  useEffect(() => {
    const fetchQuota = async () => {
      try {
        const response = await axios.get(`/api/workspaces/${workspaceId}/quota`);
        setQuota(response.data);
      } catch (err) {
        setError('Failed to fetch quota settings');
      }
    };
    fetchQuota();
  }, [workspaceId]);

  const handleSubmit = async (e) => {
    e.preventDefault();
    try {
      await axios.put(`/api/workspaces/${workspaceId}/quota`, quota);
      setSuccess(true);
      setError(null);
    } catch (err) {
      setError('Failed to update quota settings');
      setSuccess(false);
    }
  };

  const handleChange = (e) => {
    const { name, value } = e.target;
    setQuota(prev => ({ ...prev, [name]: value }));
  };

  return (
    <Card>
      <Card.Header>Workspace Quota Settings</Card.Header>
      <Card.Body>
        {error && <Alert variant="danger">{error}</Alert>}
        {success && <Alert variant="success">Quota settings updated successfully!</Alert>}
        <Form onSubmit={handleSubmit}>
          <Form.Group controlId="formCpu">
            <Form.Label>CPU Cores</Form.Label>
            <Form.Control
              type="number"
              name="cpu"
              value={quota.cpu}
              onChange={handleChange}
              placeholder="Enter CPU cores"
            />
          </Form.Group>
          <Form.Group controlId="formRam">
            <Form.Label>RAM (GB)</Form.Label>
            <Form.Control
              type="number"
              name="ram"
              value={quota.ram}
              onChange={handleChange}
              placeholder="Enter RAM in GB"
            />
          </Form.Group>
          <Form.Group controlId="formStorage">
            <Form.Label>Storage (GB)</Form.Label>
            <Form.Control
              type="number"
              name="storage"
              value={quota.storage}
              onChange={handleChange}
              placeholder="Enter Storage in GB"
            />
          </Form.Group>
          <Button variant="primary" type="submit">
            Update Quota
          </Button>
        </Form>
      </Card.Body>
    </Card>
  );
};

export default QuotaUI;
