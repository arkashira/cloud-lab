# tests/test_dockerfile_generator.py

import unittest
from unittest.mock import patch
from cloud_lab.dockerfile_generator import DockerfileGenerator

class TestDockerfileGenerator(unittest.TestCase):
    @patch('builtins.input', side_effect=['python', '3.9', 'ubuntu', 'latest'])
    def test_generate_dockerfile(self, mock_input):
        generator = DockerfileGenerator()
        dockerfile = generator.generate_dockerfile()
        self.assertIn('FROM python:3.9', dockerfile)
        self.assertIn('RUN apt-get update', dockerfile)
        self.assertIn('COPY . /app', dockerfile)
        self.assertIn('CMD ["python", "app.py"]', dockerfile)

    def test_generate_dockerfile_with_custom_base_image(self):
        generator = DockerfileGenerator()
        dockerfile = generator.generate_dockerfile(base_image='node:14')
        self.assertIn('FROM node:14', dockerfile)

    def test_generate_dockerfile_with_custom_directives(self):
        generator = DockerfileGenerator()
        dockerfile = generator.generate_dockerfile(directives=['EXPOSE 8080', 'WORKDIR /app'])
        self.assertIn('EXPOSE 8080', dockerfile)
        self.assertIn('WORKDIR /app', dockerfile)

    def test_generate_dockerfile_with_preview(self):
        generator = DockerfileGenerator()
        dockerfile = generator.generate_dockerfile(preview=True)
        self.assertIn('Generated Dockerfile:', dockerfile)

if __name__ == '__main__':
    unittest.main()
