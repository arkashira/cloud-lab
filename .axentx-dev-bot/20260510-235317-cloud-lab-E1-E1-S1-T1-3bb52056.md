// distribution.test.js
import { Distribution } from '@/models/distribution';
import { SandboxStorage } from '@/storage/sandbox';

describe('Distribution Model Unit Tests', () => {
  beforeEach(() => {
    SandboxStorage.clear();
  });

  test('create() throws ValueError for non‑string domain', () => {
    expect(() => Distribution.create(123)).toThrow(ValueError);
    expect(() => Distribution.create(null)).toThrow(ValueError);
  });

  test('create() throws ValueError for invalid domain format', () => {
    const bad = ['-bad.domain', 'bad..domain', 'toolong'.repeat(10) + '.com'];
    bad.forEach(d => expect(() => Distribution.create(d)).toThrow(ValueError));
  });

  test('created instance has correctly formatted id', () => {
    const dist = Distribution.create('example.com');
    expect(dist.id).toMatch(/^dist-[0-9a-f]{12}$/);
  });

  test('instance persists to SandboxStorage with matching fields', () => {
    const domain = 'myapp.example.org';
    const dist = Distribution.create(domain);
    const stored = SandboxStorage.get(dist.id);
    expect(stored).toBeDefined();
    expect(stored.id).toBe(dist.id);
    expect(stored.custom_domain).toBe(domain);
  });

  test('is_valid_domain returns true for valid FQDNs', () => {
    const valid = ['example.com', 'sub.domain.co.uk', 'a-b.c'];
    valid.forEach(d => expect(Distribution.is_valid_domain(d)).toBe(true));
  });

  test('is_valid_domain returns false for invalid strings', () => {
    const invalid = ['-bad.domain', 'bad..domain', 'toolong'.repeat(10) + '.com', ''];
    invalid.forEach(d => expect(Distribution.is_valid_domain(d)).toBe(false));
  });

  test('SandboxStorage.clear empties the store', () => {
    Distribution.create('example.com');
    expect(Object.keys(SandboxStorage.all()).length).toBeGreaterThan(0);
    SandboxStorage.clear();
    expect(SandboxStorage.all()).toEqual({});
  });
});
