import json
import re
import uuid
import pathlib
from threading import Lock
from typing import Dict, Any

# Thread-safe simple file-backed storage for simulated resources
_STORAGE_PATH = pathlib.Path("/opt/axentx/cloud-lab/data/distributions.json")
_STORAGE_LOCK = Lock()

_DOMAIN_REGEX = re.compile(
    r"^(?:[a-zA-Z0-9]"          # First character of the domain
    r"(?:[a-zA-Z0-9-]{0,61}[a-zA-Z0-9])?\.)+"  # Sub domain + dot
    r"[a-zA-Z]{2,}$"            # Top level domain
)

def _load_storage() -> Dict[str, Any]:
    """Load the JSON storage file, creating it if missing."""
    with _STORAGE_LOCK:
        if not _STORAGE_PATH.exists():
            _STORAGE_PATH.parent.mkdir(parents=True, exist_ok=True)
            _STORAGE_PATH.write_text(json.dumps({}))
        try:
            return json.loads(_STORAGE_PATH.read_text())
        except json.JSONDecodeError:
            # Corrupt file – reset
            _STORAGE_PATH.write_text(json.dumps({}))
            return {}

def _save_storage(data: Dict[str, Any]) -> None:
    """Persist the entire storage dict to disk."""
    with _STORAGE_LOCK:
        _STORAGE_PATH.write_text(json.dumps(data, indent=2))

def create_distribution(custom_domain: str) -> Dict[str, Any]:
    """
    Create a simulated CloudFront distribution.

    Args:
        custom_domain: The custom domain to associate with the distribution.
                       Must match a valid DNS domain pattern.

    Returns:
        A dict containing the generated ``distribution_id`` and the ``custom_domain``.

    Raises:
        ValueError: If ``custom_domain`` is not a valid domain.
    """
    if not isinstance(custom_domain, str) or not _DOMAIN_REGEX.match(custom_domain):
        raise ValueError("Invalid domain format")

    distribution_id = f"sim-{uuid.uuid4().hex[:12]}"
    record = {
        "distribution_id": distribution_id,
        "custom_domain": custom_domain,
    }

    storage = _load_storage()
    storage[distribution_id] = record
    _save_storage(storage)

    return record
```
