from enum import Enum
from dataclasses import dataclass, field
from typing import Optional

class DistributionStatus(str, Enum):
    """Enumeration of possible distribution lifecycle states.

    The values are stored as strings to simplify JSON serialization and database storage.
    The SIMULATED state is used for learner-focused distributions that do not correspond
    to a real CDN deployment.
    """
    SIMULATED = "simulated"
    ACTIVE = "active"
    DECOMMISSIONED = "decommissioned"

class DistributionState(Enum):
    """Enum representing the state of a distribution."""
    PENDING = "pending"
    ACTIVE = "active"
    DELETED = "deleted"

@dataclass
class DistributionMetrics:
    """Metrics for a distribution.

    Placeholder metrics for a distribution. Real implementations will replace these
    with live data collection.
    """
    latency_ms: Optional[float] = None  # Average latency in milliseconds
    cache_hit_ratio: Optional[float] = None  # Ratio between 0.0 and 1.0

@dataclass
class Distribution:
    """Core model representing a CDN distribution within the cloud-lab environment."""
    id: str
    name: str
    status: DistributionStatus = DistributionStatus.SIMULATED
    state: DistributionState = DistributionState.PENDING
    metrics: DistributionMetrics = field(default_factory=DistributionMetrics)

    def is_simulated(self) -> bool:
        """Convenient helper to check if the distribution is in the simulated state."""
        return self.status == DistributionStatus.SIMULATED

    def is_active(self) -> bool:
        """Convenient helper to check if the distribution is active."""
        return self.state == DistributionState.ACTIVE

    def to_dict(self) -> dict:
        """Serialize the distribution to a plain dict for API responses."""
        return {
            "id": self.id,
            "name": self.name,
            "status": self.status.value,
            "state": self.state.value,
            "metrics": {
                "latency_ms": self.metrics.latency_ms,
                "cache_hit_ratio": self.metrics.cache_hit_ratio,
            },
        }
