## 1. Acceptance Criteria
- **Real‑time cost estimate**: The algorithm must return a cost estimate within 200 ms for a sandbox with up to 10 VMs and 5 storage volumes.
- **Threshold alert**: When the estimated cost exceeds the user‑defined threshold (e.g., $50), an alert event is emitted with the exact over‑run amount.
- **Dashboard integration**: The estimate is exposed via a REST endpoint `/api/sandbox/:id/cost` and is displayed on the sandbox dashboard without additional latency (> 300 ms).
- **Accuracy**: The estimate must be within ±5 % of the actual billed amount for 95 % of test cases.
- **Error handling**: Invalid inputs (negative VM count, unsupported region) must return a 400 error with a clear message.
- **Scalability**: The algorithm should handle concurrent requests (≥ 100 QPS) without degradation beyond 10 % of baseline latency.

## 2. Unit Tests (pseudo‑code, Jest style)

// cost.test.js
const { estimateCost, validateInput } = require('./cost');

describe('validateInput', () => {
  test('accepts valid sandbox config', () => {
    const cfg = { vms: 3, storage: 2, region: 'us-east-1' };
    expect(() => validateInput(cfg)).not.toThrow();
  });

  test('throws on negative VM count', () => {
    const cfg = { vms: -1, storage: 1, region: 'us-east-1' };
    expect(() => validateInput(cfg)).toThrow('Invalid VM count');
  });

  test('throws on unsupported region', () => {
    const cfg = { vms: 1, storage: 1, region: 'unknown-region' };
    expect(() => validateInput(cfg)).toThrow('Unsupported region');
  });
});

describe('estimateCost', () => {
  test('returns cost for 1 VM, 1 GB storage in us-east-1', () => {
    const cfg = { vms: 1, storage: 1, region: 'us-east-1' };
    const cost = estimateCost(cfg);
    expect(cost).toBeCloseTo(0.05, 2); // example price
  });

  test('cost scales linearly with VM count', () => {
    const cfg1 = { vms: 1, storage: 0, region: 'us-east-1' };
    const cfg2 = { vms: 2, storage: 0, region: 'us-east-1' };
    expect(estimateCost(cfg2)).toBeCloseTo(2 * estimateCost(cfg1), 2);
  });

  test('throws on invalid config', () => {
    const cfg = { vms: 0, storage: -1, region: 'us-east-1' };
    expect(() => estimateCost(cfg)).toThrow();
  });
});
