// /opt/axentx/cloud-lab/server/websocket.js
/**
 * WebSocket Server for Real-time Cloud Lab Sandbox Sync
 * 
 * Features:
 * - Multi-tenant sandbox rooms
 * - User presence indicators
 * - Resource sync with last-write-wins conflict resolution
 * - Broadcast batching (within 1 second)
 */

const WebSocket = require('ws');
const http = require('http');
const url = require('url');
const crypto = require('crypto');

const PORT = process.env.WS_PORT || 8080;
const BROADCAST_DELAY_MS = 100;

// In-memory storage (production: use Redis)
const sandboxes = new Map(); // sandboxId -> { clients, resources, lastUpdate }
const clientSessions = new Map(); // ws -> session data

const server = http.createServer((req, res) => {
  res.writeHead(200, { 'Content-Type': 'application/json' });
  res.end(JSON.stringify({ status: 'ok', service: 'cloud-lab-websocket' });
});

const wss = new WebSocket.Server({ server });

// ═══════════════════════════════════════════════════════════════
// HELPERS
// ═══════════════════════════════════════════════════════════════

function generateMessageId() {
  return crypto.randomUUID();
}

function getSandbox(sandboxId) {
  if (!sandboxes.has(sandboxId)) {
    sandboxes.set(sandboxId, {
      clients: new Map(), // ws -> { userId, username, joinedAt }
      resources: new Map(), // resourceId -> { data, lastModified, modifiedBy }
      lastUpdate: Date.now()
    });
  }
  return sandboxes.get(sandboxId);
}

function sendToClient(ws, message) {
  if (ws.readyState === WebSocket.OPEN) {
    ws.send(JSON.stringify({ ...message, messageId: generateMessageId(), timestamp: Date.now() });
  }
}

function broadcastToSandbox(sandboxId, message, excludeWs = null) {
  const sandbox = sandboxes.get(sandboxId);
  if (!sandbox) return;

  const messageStr = JSON.stringify(message);
  
  sandbox.clients.forEach((clientInfo, ws) => {
    if (ws !== excludeWs && ws.readyState === WebSocket.OPEN) {
      ws.send(messageStr);
    }
  });
}

function broadcastWithDelay(sandboxId, message, excludeWs = null) {
  setTimeout(() => broadcastToSandbox(sandboxId, message, excludeWs), BROADCAST_DELAY_MS);
}

// ═══════════════════════════════════════════════════════════════
// MESSAGE HANDLERS
// ═══════════════════════════════════════════════════════════════

function handleResourceUpdate(ws, message, session) {
  const { sandboxId, userId, username } = session;
  const sandbox = getSandbox(sandboxId);
  const { resourceId, data } = message;

  const existing = sandbox.resources.get(resourceId);
  const lastModified = Date.now();

  // Last-write-wins conflict resolution
  if (existing && existing.lastModified > lastModified) {
    // Notify sender their update was superseded
    sendToClient(ws, {
      type: 'conflict',
      resourceId,
      message: 'Resource was modified by another user',
      currentVersion: existing
    });
    return;
  }

  sandbox.resources.set(resourceId, {
    data,
    lastModified,
    modifiedBy: userId,
    version: (existing?.version || 0) + 1
  });

  // Broadcast to others with delay (batch multiple rapid updates)
  broadcastWithDelay(sandboxId, {
    type: 'resource_update',
    resourceId,
    data,
    lastModified,
    modifiedBy: userId,
    modifiedByName: username
  }, ws);
}
