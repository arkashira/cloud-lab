const WebSocket = require('ws');
const { EventEmitter } = require('events');
const crypto = require('crypto');

class SandboxSyncService extends EventEmitter {
  constructor(options = {}) {
    super();
    this.wsServer = null;
    this.clients = new Map(); // clientId -> { socket, presence, lastWrite }
    this.sandboxState = new Map(); // resourceId -> { data, version, lastModified, author }
    this.changeQueue = [];
    this.broadcastTimeout = 1000;
    this.conflictResolver = options.conflictResolver || this.lastWriteWins;
    this.eventEmitter = new EventEmitter();
  }

  start(port = 8080) {
    this.wsServer = new WebSocket.Server({ port });
    this.wsServer.on('connection', (socket) => {
      this.handleConnection(socket);
    });
    console.log(`Sandbox sync server listening on port ${port}`);
  }

  stop() {
    if (this.wsServer) {
      this.wsServer.close();
    }
  }

  handleConnection(socket) {
    const clientId = crypto.randomUUID();
    const client = {
      socket,
      presence: {
        joinedAt: Date.now(),
        lastActive: Date.now(),
        name: socket.handshake.query.name || 'Anonymous'
      },
      lastWrite: null,
      pendingChanges: []
    };

    this.clients.set(clientId, client);
    socket.id = clientId;

    // Send current state to new client
    this.broadcastState(clientId);

    // Send presence updates to all other clients
    this.broadcastPresence(clientId, 'joined');

    socket.on('message', (data) => {
      this.handleMessage(clientId, data);
    });

    socket.on('close', () => {
      this.handleDisconnect(clientId);
    });

    socket.on('error', (err) => {
      console.error(`Socket error for ${clientId}:`, err);
      this.handleDisconnect(clientId);
    });
  }

  handleMessage(clientId, data) {
    const client = this.clients.get(clientId);
    if (!client) return;

    try {
      const message = JSON.parse(data.toString());
      const { type, resourceId, data: payload, version } = message;

      switch (type) {
        case 'resource-add':
          this.handleResourceAdd(clientId, resourceId, payload, version);
          break;
        case 'resource-update':
          this.handleResourceUpdate(clientId, resourceId, payload, version);
          break;
        case 'resource-delete':
          this.handleResourceDelete(clientId, resourceId);
          break;
        case 'heartbeat':
          client.presence.lastActive = Date.now();
          break;
        default:
          console.warn(`Unknown message type: ${type}`);
      }
    } catch (err) {
      console.error(`Invalid message from ${clientId}:`, err);
    }
  }

  handleResourceAdd(clientId, resourceId, payload, version) {
    const existing = this.sandboxState.get(resourceId);
    const conflict = existing && existing.version >= version;

    if (conflict) {
      this.emit('conflict', {
        clientId,
        resourceId,
        existing: existing,
        incoming: { payload, version }

      return;
