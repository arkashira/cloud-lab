describe('roles', () => {
  describe('createSandbox', () => {
    it('returns a unique URL and password', () => {
      const sandbox = createSandbox();
      expect(sandbox.url).toBeA('string');
      expect(sandbox.password).toBeA('string');
    });

    it('stores URL and password in the database', () => {
      const sandbox = createSandbox();
      const storedSandbox = getSandboxesFromDatabase();
      expect(storedSandbox.url).toBe(sandbox.url);
      expect(storedSandbox.password).toBe(sandbox.password);
    });
  });

  describe('joinSandbox', () => {
    it('accepts different roles', () => {
      const sandbox = createSandbox();
      const user1 = joinSandbox(sandbox.url, 'admin');
      const user2 = joinSandbox(sandbox.url, 'contributor');
      expect(user1.role).toBe('admin');
      expect(user2.role).toBe('contributor');
    });

    it('users with different roles have different permissions', () => {
      const sandbox = createSandbox();
      const user1 = joinSandbox(sandbox.url, 'admin');
      const user2 = joinSandbox(sandbox.url, 'contributor');
      expect(user1.hasPermission('manageTeam')).toBe(true);
      expect(user2.hasPermission('manageTeam')).toBe(false);
    });
  });

  describe('manageTeam', () => {
    it('allows admins to add and remove team members', () => {
      const sandbox = createSandbox();
      const user1 = joinSandbox(sandbox.url, 'admin');
      const user2 = joinSandbox(sandbox.url, 'contributor');
      user1.addTeamMember(user2);
      expect(user1.getTeamMembers()).toContain(user2);
      user1.removeTeamMember(user2);
      expect(user1.getTeamMembers()).not.toContain(user2);
    });

    it('allows admins to update team member roles', () => {
      const sandbox = createSandbox();
      const user1 = joinSandbox(sandbox.url, 'admin');
      const user2 = joinSandbox(sandbox.url, 'contributor');
      user1.updateTeamMemberRole(user2, 'admin');
      expect(user2.role).toBe('admin');
    });
  });
});
