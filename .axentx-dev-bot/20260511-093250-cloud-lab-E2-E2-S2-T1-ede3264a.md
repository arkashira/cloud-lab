const { MongoClient } = require('mongodb');
const client = new MongoClient('mongodb://localhost:27017', { useNewUrlParser: true, useUnifiedTopology: true });
const dbName = 'cloud-lab';

async function connectToDatabase() {
  await client.connect();
  console.log('Connected successfully to server');
  return client.db(dbName);
}

async function setCostCappingLimit(userId, limit) {
  const db = await connectToDatabase();
  const collection = db.collection('users');
  await collection.updateOne({ _id: userId }, { $set: { costCapLimit: limit } });
}

async function scheduleCostCapping(userId, schedule) {
  const db = await connectToDatabase();
  const collection = db.collection('users');
  await collection.updateOne({ _id: userId }, { $set: { costCapSchedule: schedule } });
}

async function getCostCappingHistory(userId) {
  const db = await connectToDatabase();
  const collection = db.collection('users');
  const user = await collection.findOne({ _id: userId });
  return user.costCapHistory || [];
}

module.exports = {
  setCostCappingLimit,
  scheduleCostCapping,
  getCostCappingHistory
};
