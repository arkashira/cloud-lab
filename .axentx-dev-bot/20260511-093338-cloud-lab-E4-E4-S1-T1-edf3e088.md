//  Skill‑Badges Service – Express router + helper

//  • Admin‑only settings endpoints

//  • awardBadge(userId, badgeId) helper for other modules

const fs = require('fs');
const path = require('path');

/* ------------------------------------------------------------------
   1️⃣  CONFIGURATION & PERSISTENCE
   ------------------------------------------------------------------ */
// Path for optional JSON persistence (you can delete the block if you prefer pure memory)
const PERSISTENCE_FILE = path.join(__dirname, '..', 'data', 'skill-badges.json');

// Default configuration – will be overwritten by persisted data if present
const badgeSettings = {
  enabled: true,
  badges: [
    {
      id: 'deployment_master',
      name: 'Deployment Master',
      description: 'Completed a mock‑deployment scenario',
    },
  ],
};

// Load persisted state (if any)
(function loadPersisted() {
  try {
    const raw = fs.readFileSync(PERSISTENCE_FILE, 'utf8');
    const persisted = JSON.parse(raw);
    if (typeof persisted.enabled === 'boolean') badgeSettings.enabled = persisted.enabled;
    if (Array.isArray(persisted.badges)) badgeSettings.badges = persisted.badges;
    if (persisted.history) userBadgeHistory = new Map(persisted.history);
    console.info('[skill‑badges] Loaded persisted state.');
  } catch (_) {
    // No persisted file – start fresh
  }
})();

// Helper to write the whole state back to disk (called after every mutation)
function persistState() {
  try {
    const dir = path.dirname(PERSISTENCE_FILE);
    if (!fs.existsSync(dir)) fs.mkdirSync(dir, { recursive: true });
    const data = {
      enabled: badgeSettings.enabled,
      badges: badgeSettings.badges,
      // Convert Map → Array for JSON
      history: Array.from(userBadgeHistory.entries()),
    };
    fs.writeFileSync(PERSISTENCE_FILE, JSON.stringify(data, null, 2));
  } catch (e) {
    console.error('[skill‑badges] Persistence error:', e);
  }
}
