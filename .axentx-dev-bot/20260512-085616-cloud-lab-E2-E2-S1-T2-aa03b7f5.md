#!/usr/bin/env bash
# /opt/axentx/cloud-lab/scripts/test.sh
#
# Purpose: Verify that the Kubernetes cluster provisioned by Terraform
#          is reachable and that all nodes are in the Ready state.
#          Produce a JSON report for downstream consumption.
#
# Requirements:
#   - kubectl (>=1.18)
#   - jq
#   - KUBECONFIG env var or $HOME/.kube/config

set -euo pipefail

# ------------------------------------------------------------------
# Logging helper
# ------------------------------------------------------------------
log() {
  printf '[%s] %s\n' "$(date -u '+%Y-%m-%dT%H:%M:%SZ')" "$*"
}

# ------------------------------------------------------------------
# 1. Prerequisites
# ------------------------------------------------------------------
# kubectl must be on PATH
if ! command -v kubectl >/dev/null 2>&1; then
  log "ERROR: kubectl not found in PATH."
  exit 1
fi

# jq must be on PATH
if ! command -v jq >/dev/null 2>&1; then
  log "ERROR: jq not found in PATH."
  exit 1
fi

# KUBECONFIG – either env var or default location
if [[ -z "${KUBECONFIG:-}" ]]; then
  if [[ -f "$HOME/.kube/config" ]]; then
    export KUBECONFIG="$HOME/.kube/config"
    log "Using default kubeconfig at $KUBECONFIG"
  else
    log "ERROR: KUBECONFIG is not set and no default kubeconfig found."
    exit 1
  fi
fi

# ------------------------------------------------------------------
# 2. Connectivity check
# ------------------------------------------------------------------
log "Checking kubectl client version..."
kubectl version --client --short | grep -q 'Client Version' || { log "ERROR: kubectl client check failed."; exit 1; }

log "Verifying API server reachability..."
if ! kubectl version --short --client; then
  log "ERROR: Unable to communicate with the Kubernetes cluster."
  exit 1
fi

# ------------------------------------------------------------------
# 3. Node health
# ------------------------------------------------------------------
log "Retrieving node list..."
NODES_JSON=$(kubectl get nodes -o json)

NODE_COUNT=$(echo "$NODES_JSON" | jq -r '.items | length')
if [[ "$NODE_COUNT" -eq 0 ]]; then
  log "ERROR: No nodes found in the cluster."
  exit 1
fi

READY_COUNT=$(echo "$NODES_JSON" | jq -r '
  .items[]
  | select(.status.conditions[] | select(.type=="Ready" and .status=="True"))
  | .metadata.name
' | wc -l)

NODE_NAMES=$(echo "$NODES_JSON" | jq -r '.items[].metadata.name' | paste -sd "," -)

if [[ "$READY_COUNT" -lt "$NODE_COUNT" ]]; then
  log "WARNING: Only $READY_COUNT of $NODE_COUNT nodes are Ready."
else
  log "All $NODE_COUNT nodes are Ready."
fi

# ------------------------------------------------------------------
# 4. Report
# ----------------------------------------

  - `kubectl get nodes -o json` returns at least one node.

  - The JSON contains:  
    - `context`: current kubeconfig context name.  
    - `total_nodes`: integer count of all nodes.  
    - `ready_nodes`: integer count of nodes with status `Ready`.  
    - `node_names`: comma‑separated string of node names.

- **Logging**  
  - Every log line starts with a UTC ISO‑8601 timestamp (`YYYY-MM-DDTHH:MM:SSZ`).

- **Makefile target**  
  - `make test` makes `/opt/axentx/cloud-lab/scripts/test.sh` executable and runs it.
