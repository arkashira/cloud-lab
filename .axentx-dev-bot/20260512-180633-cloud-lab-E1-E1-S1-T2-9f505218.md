// /opt/axentx/cloud-lab/frontend/components/services/AzureServices.js
import React, { useState } from 'react';
import axios from 'axios';

const AzureServices = () => {
  const [selectedService, setSelectedService] = useState('');
  const [services, setServices] = useState([]);

  const fetchAzureServices = async () => {
    try {
      const response = await axios.get('/api/azure/services');
      setServices(response.data);
    } catch (error) {
      console.error('Error fetching Azure services:', error);
    }
  };

  const handleServiceSelection = (event) => {
    setSelectedService(event.target.value);
  };

  const createResource = async () => {
    try {
      await axios.post('/api/azure/resources', { service: selectedService });
      alert('Resource created successfully!');
    } catch (error) {
      console.error('Error creating Azure resource:', error);
      alert('Failed to create resource.');
    }
  };

  const deleteResource = async () => {
    try {
      await axios.delete(`/api/azure/resources/${selectedService}`);
      alert('Resource deleted successfully!');
    } catch (error) {
      console.error('Error deleting Azure resource:', error);
      alert('Failed to delete resource.');
    }
  };

  return (
    <div>
      <h2>Azure Services</h2>
      <button onClick={fetchAzureServices}>Fetch Services</button>
      <select value={selectedService} onChange={handleServiceSelection}>
        <option value="">Select a service</option>
        {services.map((service) => (
          <option key={service.id} value={service.name}>
            {service.name}
          </option>
        ))}
      </select>
      <button onClick={createResource}>Create Resource</button>
      <button onClick={deleteResource}>Delete Resource</button>
    </div>
  );
};

export default AzureServices;
```

- [ ] Upon selecting Azure, the UI displays relevant Azure resources and options.
- [ ] Users can navigate away from the Azure service selection page and return later without losing their selection.
- [ ] The selected service provider (Azure) is persisted across sessions.
- [ ] Users can switch to a different service provider (e.g., AWS, GCP) if desired.

  it('displays Azure as selected service provider', () => {
    const { getByText } = render(<AzureServices />);
    expect(getByText('Azure')).toBeInTheDocument();
  });

  it('persists selected service provider across sessions', () => {
    const { getByText, queryByText } = render(<AzureServices />);
    fireEvent.click(getByText('Switch to AWS'));
    expect(queryByText('Azure')).not.toBeInTheDocument();
    // Clear local storage and rerender to simulate a new session
    localStorage.clear();
    render(<AzureServices />);
    expect(queryByText('Azure')).not.toBeInTheDocument();
  });
});
```

**3. Integration Tests**

*Happy Paths*

- Verify that selecting Azure as the service provider redirects the user to the Azure resources page.
- Ensure that the user can switch back to another service provider (e.g., AWS) from the Azure resources page.
- Confirm that the selected service provider is persisted across sessions.

*Edge Cases*

- Verify that attempting to select an unsupported service provider (e.g., 'Other') displays an error message.
- Ensure that the UI handles network errors gracefully when trying to fetch service provider data.
