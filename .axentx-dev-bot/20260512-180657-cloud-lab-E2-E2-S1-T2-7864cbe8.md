  const secret = 'test-secret';
  const token = jwt.sign({ sub: 'bob@example.com', roles: ['azure:read'] }, secret);

  test('allows access when required role is present', () => {
    const result = verifyJwtRoles(token, secret, 'azure:read');
    expect(result).toBe(true);
  });

  test('denies access when required role is missing', () => {
    const result = verifyJwtRoles(token, secret, 'azure:write');
    expect(result).toBe(false);

  test('denies access for malformed token', () => {

  });
