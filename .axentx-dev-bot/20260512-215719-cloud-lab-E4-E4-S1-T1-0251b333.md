# Deploy stage: builds, pushes, and deploys the application
deploy:
  stage: deploy
  script:
    - docker build -t $DOCKER_IMAGE .  # Build the Docker image
    - docker push $DOCKER_IMAGE  # Push the Docker image to the registry
    - kubectl apply -f kubernetes/deployment.yaml  # Apply the Kubernetes deployment
  only:
    - main  # Restrict deployment to the main branch
```
