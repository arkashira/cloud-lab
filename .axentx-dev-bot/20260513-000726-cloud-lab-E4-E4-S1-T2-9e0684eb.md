#!/usr/bin/env python3
# /opt/axentx/cloud-lab/monitoring/grafana_setup.py
"""
Grafana dashboard import utility.

This script is intended for the cloud‑lab provisioning process.  
It can be invoked either as a command‑line tool or via environment
variables (useful when the script is called from a Docker entrypoint).

Usage (CLI):
    python grafana_setup.py \
        --url http://grafana.local:3000 \
        --api-key <GRAFANA_API_KEY> \
        --dashboard /path/to/dashboard.json

Environment‑variable fallback (no CLI arguments required):
    GRAFANA_URL      – Base URL of the Grafana instance
    GRAFANA_API_KEY  – API key with Admin or Editor permissions
    DASHBOARD_JSON   – Path to the dashboard JSON file

The script will:
* Load the JSON file, validating it and wrapping it in the
  structure expected by `/api/dashboards/db`.
* POST the payload to Grafana, overwriting an existing dashboard
  if one with the same UID already exists.
* Log progress and any errors, exiting with a non‑zero status on failure.
"""

from __future__ import annotations

import argparse
import json
import logging
import os
import sys
from pathlib import Path
from typing import Any, Dict

import requests

# --------------------------------------------------------------------------- #
# Logging configuration
# --------------------------------------------------------------------------- #
logging.basicConfig(
    level=logging.INFO,
    format="%(levelname)s: %(message)s",
)
LOGGER = logging.getLogger(__name__)

# --------------------------------------------------------------------------- #
# Constants
# --------------------------------------------------------------------------- #
GRAFANA_API_ENDPOINT = "/api/dashboards/db"
DEFAULT_TIMEOUT = 10  # seconds


# --------------------------------------------------------------------------- #
# Helper functions
# --------------------------------------------------------------------------- #
def load_dashboard_json(path: Path) -> Dict[str, Any]:
    """
    Load a Grafana dashboard JSON file and normalise it for the API.

    Parameters
    ----------
    path : Path
        Path to the JSON file.

    Returns
    -------
    dict
        Payload ready for the `/api/dashboards/db` endpoint.

    Raises
    ------
    FileNotFoundError
        If the file does not exist.
    ValueError
        If the file contains invalid JSON.
    """
    if not path.is_file():
        raise FileNotFoundError(f"Dashboard JSON not found: {path}")

    with path.open("r", encoding="utf-8") as f:
        try:
            data = json.load(f)
        except json.JSONDecodeError as exc:
            raise ValueError(f"Invalid JSON in {path}: {exc}") from exc

    # Grafana expects the top‑level object to contain a "dashboard" key
    # when using the /api/dashboards/db endpoint.  If the file already
    # follows that schema we keep it, otherwise we wrap it.
    if "dashboard" not in data:
        data = {"dashboard": data, "overwrite": True, "folderId": 0}
    else:
        # Ensure required fields are present
        data.setdefault("overwrite", True)
        data.setdefault("folderId", 0)

    return data


def import_dashboard(grafana_url: str, api_key: str, payload: Dict[str, Any]) -> None:
    """
    POST the dashboard JSON to Grafana's /api/dashboards/db endpoint.

    Parameters
    ----------
    grafana_url : str
        Base URL of the Grafana instance (e.g. http://localhost:3000).
    api_key : str
        Grafana API key with Admin or Editor permissions.
    payload : dict
        Normalised dashboard payload.

    Raises
    ------
    requests.HTTPError
        If Grafana returns a non‑2xx response.
    RuntimeError
        If Grafana reports a failure in the JSON response.
    """
    endpoint = f"{grafana_url.rstrip('/')}{GRAFANA_API_ENDPOINT}"
    headers = {
        "Authorization": f"Bearer {api_key}",
        "Content-Type": "application/json",
    }
