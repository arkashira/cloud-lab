def _run_ansible_async(job_id: str, playbook_path: str):
    """Run Ansible playbook in background thread."""
    try:
        _update_job_status(job_id, ProvisionStatus.ANSIBLE_RUNNING, "Starting Ansible playbook")
        _append_log(job_id, "info", "Ansible playbook initiated")

        ansible_service = AnsibleService()
        result = ansible_service.run_playbook(playbook_path)

        if result.get("success"):
            _update_job_status(
                job_id,
                ProvisionStatus.ANSIBLE_COMPLETE,
                "Ansible playbook completed successfully",
                {"ansible_output": result}
            )
            _append_log(job_id, "info", "Ansible playbook completed successfully")
            _append_log(job_id, "ansible", result.get("logs", ""))
            return True
        else:
            _update_job_status(
                job_id,
                ProvisionStatus.FAILED,
                "Ansible playbook failed",
                {"error": result.get("logs")}
            )
            _append_log(job_id, "error", f"Ansible playbook failed: {result.get('logs')}")
            return False
    except Exception as e:
        _update_job_status(
            job_id,
            ProvisionStatus.FAILED,
            "Ansible playbook failed with exception",
            {"error": str(e)}
        )
        _append_log(job_id, "error", f"Ansible playbook failed with exception: {str(e)}")
        return False

@provision_bp.route('/provision', methods=['POST'])
def provision():
    """Trigger provisioning workflow."""
    data = request.get_json()
    config_path = data.get('config_path')
    playbook_path = data.get('playbook_path')
    workspace = data.get('workspace', 'default')

    if not config_path or not playbook_path:
        return jsonify({"error": "config_path and playbook_path are required"}), 400

    job_id = str(int(time.time() * 1000))
    _update_job_status(job_id, ProvisionStatus.PENDING, "Provisioning job created")

    # Run Terraform in background
    threading.Thread(
        target=_run_terraform_async,
        args=(job_id, config_path, workspace),
        daemon=True
    ).start()

    return jsonify({
        "job_id": job_id,
        "status": ProvisionStatus.PENDING,
        "message": "Provisioning job created"
    })

@provision_bp.route('/provision/<job_id>/status', methods=['GET'])
def get_provision_status(job_id: str):
    """Get status of a provisioning job."""
    with _provision_lock:
        if job_id not in _provision_jobs:
            return jsonify({"error": "Job not found"}), 404
        return jsonify(_provision_jobs[job_id])

@provision_bp.route('/provision/<job_id>/logs', methods=['GET'])
def get_provision_logs(job_id: str):
    """Get logs for a provisioning job."""
    with _provision_lock:
        if job_id not in _provision_jobs:
            return jsonify({"error": "Job not found"}), 404
        return jsonify({
            "job_id": job_id,
            "logs": _provision_jobs[job_id]["logs"]
        })
