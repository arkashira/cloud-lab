import React from 'react';
import { render, fireEvent, waitFor } from '@testing-library/react';
import { TemplateCatalog } from './TemplateCatalog';
import { mockTemplates } from './mockTemplates';

describe('TemplateCatalog', () => {
  it('renders a card for each template', () => {
    const { getByTestId } = render(<TemplateCatalog templates={mockTemplates} />);
    expect(getByTestId('template-card')).toBeInTheDocument();
    expect(getByTestId('template-card')).toHaveTextContent(mockTemplates[0].name);
  });

  it('filters by search term', async () => {
    const { getByPlaceholderText, getByText } = render(<TemplateCatalog templates={mockTemplates} />);
    const searchInput = getByPlaceholderText('Search');
    fireEvent.change(searchInput, { target: { value: 'VPC' } });
    await waitFor(() => getByText('VPC'));
    expect(getByText('VPC')).toBeInTheDocument();
    expect(getByText('VPC')).toHaveTextContent(mockTemplates[0].name);
  });

  it('filters by component', async () => {
    const { getByRole, getByText } = render(<TemplateCatalog templates={mockTemplates} />);
    const componentCombobox = getByRole('combobox', { name: /component/i });
    fireEvent.change(componentCombobox, { target: { value: 'EKS' } });
    await waitFor(() => getByText('EKS'));
    expect(getByText('EKS')).toBeInTheDocument();
    expect(getByText('EKS')).toHaveTextContent(mockTemplates[0].name);
  });

  it('applies both search term and component filter', async () => {
    const { getByPlaceholderText, getByRole, getByText } = render(<TemplateCatalog templates={mockTemplates} />);
    const searchInput = getByPlaceholderText('Search');
    fireEvent.change(searchInput, { target: { value: 'VPC' } });
    const componentCombobox = getByRole('combobox', { name: /component/i });
    fireEvent.change(componentCombobox, { target: { value: 'S3' } });
