  <script>
    // Mobile menu toggle
    document.querySelector('.nav-toggle').addEventListener('click', function() {
      const navLinks = document.querySelector('.nav-links');

      this.setAttribute('aria-expanded', !expanded);
      navLinks.classList.toggle('open');
    });
  </script>
</body>
