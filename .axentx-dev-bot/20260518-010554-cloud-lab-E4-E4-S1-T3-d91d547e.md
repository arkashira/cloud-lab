import React from 'react';
import { render, fireEvent, waitFor } from '@testing-library/react';
import { axe, toHaveNoViolations } from 'jest-axe';
import { Provider } from 'react-redux';
import { createStore } from 'redux';
import rootReducer from '../reducers';
import AzureLabMetrics from './AzureLabMetrics';

expect.extend(toHaveNoViolations);

describe('AzureLabMetrics component', () => {
  const store = createStore(rootReducer);

  it('renders without crash and displays segment selector dropdown', () => {
    const { getByText } = render(
      <Provider store={store}>
        <AzureLabMetrics />
      </Provider>
    );
    expect(getByText('Segment')).toBeInTheDocument();
  });

  it('displays loading spinner while loading state is true', () => {
    const { getByText } = render(
      <Provider store={store}>
        <AzureLabMetrics loading={true} />
      </Provider>
    );
    expect(getByText('Loading...')).toBeInTheDocument();
  });

  it('displays error message when API call fails', async () => {
    const { getByText } = render(
      <Provider store={store}>
        <AzureLabMetrics error="API call failed" />
      </Provider>
    );
    expect(getByText('API call failed')).toBeInTheDocument();
  });

  it('chart renders with two datasets when data loads successfully', async () => {
    const { getByText } = render(
      <Provider store={store}>
        <AzureLabMetrics data={{ activeLabs: 10, newLabs: 20 }} />
      </Provider>
    );
    expect(getByText('Active Labs')).toBeInTheDocument();
    expect(getByText('New Labs')).toBeInTheDocument();
  });

  it('API call includes correct query params', async () => {
    const { getByText } = render(
      <Provider store={store}>
        <AzureLabMetrics />
      </Provider>
    );
    const axiosSpy = jest.spyOn(axios, 'get');
    const response = { data: { activeLabs: 10, newLabs: 20 } };
    axiosSpy.mockResolvedValue(response);
    await waitFor(() => expect(axiosSpy).toHaveBeenCalledTimes(1));
    expect(axiosSpy).toHaveBeenCalledWith(
      `/api/azure-lab-metrics?segment=individual&start_date=2022-01-01&end_date=2022-01-07`
    );
  });

  it('changing segment dropdown triggers a new API fetch', async () => {
    const { getByText } = render(
      <Provider store={store}>
        <AzureLabMetrics />
      </Provider>
    );
    const axiosSpy = jest.spyOn(axios, 'get');
    const response = { data: { activeLabs: 10, newLabs: 20 } };
    axiosSpy.mockResolvedValue(response);
    await waitFor(() => expect(axiosSpy).toHaveBeenCalledTimes(1));
    const segmentDropdown = getByText('Segment');
    fireEvent.change(segmentDropdown, { target: { value: 'team' } });
    await waitFor(() => expect(axiosSpy).toHaveBeenCalledTimes(2));
  });

  it('date range state initializes to current date in YYYY-MM-DD format', () => {
    const { getByText } = render(
      <Provider store={store}>
        <AzureLabMetrics />
      </Provider>
    );
    expect(getByText(new Date().toISOString().split('T')[0])).toBeInTheDocument();
  });
});
