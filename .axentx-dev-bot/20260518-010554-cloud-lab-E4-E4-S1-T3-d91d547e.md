```typescript
// src/hooks/useLabMetrics.ts
import { useState, useEffect, useMemo } from 'react';
import { LabMetric, SegmentData } from '../types/metrics';

const mockData: LabMetric[] = [
  { week: '2026-05-01', totalLabs: 120, activeLabs: 85, avgSession: 45, segment: 'individual' },
  { week: '2026-05-08', totalLabs: 145, activeLabs: 92, avgSession: 48, segment: 'team' },
  { week: '2026-05-15', totalLabs: 168, activeLabs: 105, avgSession: 52, segment: 'individual' },
  { week: '2026-05-22', totalLabs: 189, activeLabs: 120, avgSession: 55, segment: 'team' },
  { week: '2026-05-29', totalLabs: 205, activeLabs: 135, avgSession: 58, segment: 'individual' },
  { week: '2026-06-05', totalLabs: 220, activeLabs: 148, avgSession: 60, segment: 'team' },
];

const segmentDistribution: SegmentData[] = [
  { name: 'Individual', value: 65 },
  { name: 'Team', value: 35 },
];

export const useLabMetrics = () => {
  const [metrics, setMetrics] = useState<LabMetric[]>([]);
  const [filter, setFilter] = useState<string>('all');
  const [loading, setLoading] = useState<boolean>(true);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    const fetchData = async () => {
      try {
        setLoading(true);
        // Simulate API call
        await new Promise(resolve => setTimeout(resolve, 500));
        setMetrics(mockData);
      } catch (err) {
        setError('Failed to load metrics data');
      } finally {
        setLoading(false);
      }
    };

    fetchData();
  }, []);

  const filteredData = useMemo(() => {
    if (filter === 'all') return metrics;
    return metrics.filter(item => item.segment === filter);
  }, [metrics, filter]);

  const formattedData = useMemo(() => {
    return filteredData.map(item => ({
      ...item,
      date: new Date(item.week).toLocaleDateString('en-US', { month: 'short', day: 'numeric' }),
    }));
  }, [filteredData]);

  const totals = useMemo(() => {
    return {
      totalLabs: filteredData.reduce((sum, item) => sum + item.totalLabs, 0),
      activeLabs: filteredData.reduce((sum, item) => sum + item.activeLabs, 0),
      avgSession: Math.round(filteredData.reduce((sum, item) => sum + item.avgSession, 0) / filteredData.length) || 0,
    };
  }, [filteredData]);

  return {
    metrics: formattedData,
    filteredData,
    filter,
    setFilter,
    loading,
    error,
    totals,
    segmentDistribution,
  };
};
```

   - Each card must update within 2 seconds after the ETL job completes.

2. **Data Refresh**  
   - The UI must trigger a data fetch exactly once per night at 02:00 UTC.  
   - Subsequent manual refreshes (via a “Refresh” button) must re‑fetch data and update the UI within 5 seconds.

3. **User Segment Filtering**  
   - The filter component must expose two options: “Individual” and “Team”.  
   - Selecting an option must update the displayed metrics to match the selected segment within 3 seconds.  
   - The filter state must be persisted in the URL query string (e.g., `?segment=team`).

4. **Error Handling**  
   - If the API returns a non‑2xx status, the UI must display a toast with the message “Failed to load lab metrics.”  
   - The toast must disappear after 5 seconds.

5. **Loading State**  
   - While data is being fetched, a skeleton loader must be visible in place of each metric card.  
   - The loader must be removed once data arrives or an error occurs.

6. **Accessibility**  
   - All interactive elements must have accessible names (ARIA labels).  
   - The component tree must pass a WCAG 2.1 AA audit (contrast, focus order, keyboard navigation).  
   - Screen reader users must be able to identify each metric card and the filter options.

7. **Performance**  
   - Rendering the dashboard with 10,000 lab records must complete within 200 ms on a mid‑range laptop.  
   - Memory usage must not exceed 50 MB during a full refresh.

// __tests__/useLabMetrics.test.js
import { renderHook, act } from '@testing-library/react-hooks';
import { useLabMetrics } from '../../src/hooks/useLabMetrics';
import * as api from '../../src/api/labMetrics';

jest.mock('../../src/api/labMetrics');

describe('useLabMetrics', () => {
  beforeEach(() => jest.clearAllMocks());

  test('returns loading=true initially', () => {
    const { result } = renderHook(() => useLabMetrics('individual'));
    expect(result.current.loading).toBe(true);
  });

  test('fetches data on mount and sets metrics', async () => {
    const mockData = { total: 10, active: 5, avgSession: 3600 };
    api.fetchLabMetrics.mockResolvedValueOnce(mockData);

    const { result, waitForNextUpdate } = renderHook(() => useLabMetrics('individual'));

    await waitForNextUpdate(); // after fetch resolves

    expect(result.current.loading).toBe(false);
    expect(result.current.error).toBeNull();
    expect(result.current.metrics).toEqual(mockData);
  });

  test('handles API error and sets error state', async () => {
    api.fetchLabMetrics.mockRejectedValueOnce(new Error('Network'));

    const { result, waitForNextUpdate } = renderHook(() => useLabMetrics('team'));

    await waitForNextUpdate();

    expect(result.current.loading).toBe(false);
    expect(result.current.error).toBeInstanceOf(Error);
    expect(result.current.error.message).toBe('Failed to load lab metrics.');
  });

  test('refetches when segment changes', async () => {
    const data1 = { total: 8, active: 3, avgSession: 1800 };
    const data2 = { total: 12, active: 6, avgSession: 2400 };
    api.fetchLabMetrics
      .mockResolvedValueOnce(data1)
      .mockResolvedValueOnce(data2);

    const { result, rerender, waitForNextUpdate } = renderHook(
      ({ segment }) => useLabMetrics(segment),
      { initialProps: { segment: 'individual' } }
    );

    await waitForNextUpdate();
    expect(result.current.metrics).toEqual(data1);

    rerender({ segment: 'team' });
    await waitForNextUpdate();
    expect(result.current.metrics).toEqual(data2);
  });
});
