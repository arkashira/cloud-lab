/**
 * TutorialViewer component
 *

 * command via a backend endpoint. Progress is shown with a progress bar.

 */
const TutorialViewer = () => {
  // Define the tutorial steps
  const steps = [
    {
      id: 1,
      title: 'Create an Azure Resource Group',
      description: 'Create a resource group to contain all resources.',
      command: 'az group create --name myResourceGroup --location eastus',
    },
    {
      id: 2,
      title: 'Create an Azure App Service Plan',
      description: 'Provision an App Service plan for the web app.',
      command: 'az appservice plan create --name myPlan --resource-group myResourceGroup --sku B1',
    },
    {
      id: 3,
      title: 'Create an Azure Web App',
      description: 'Create the web app that will host the Node.js application.',
      command: 'az webapp create --name myNodeApp --plan myPlan --resource-group myResourceGroup --runtime "NODE|14-lts"',
    },
    {
      id: 4,
      title: 'Deploy the Node.js App',
      description: 'Deploy the local code to the Azure Web App.',
      command: 'az webapp up --name myNodeApp --resource-group myResourceGroup',
    },
    {
      id: 5,
      title: 'Verify Deployment',
      description: 'Check that the app is running and accessible.',
      command: 'az webapp browse --name myNodeApp --resource-group myResourceGroup',
    },
  ];

  const [completedSteps, setCompletedSteps] = useState([]);
  const [loadingStep, setLoadingStep] = useState(null);
  const [error, setError] = useState(null);
  const [appUrl, setAppUrl] = useState('');

  const runCommand = async (step) => {
    setLoadingStep(step.id);
    setError(null);
    
    try {
      const response = await fetch('/api/run-command', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ command: step.command }), // Fixed: was JSON.stringify{ in candidate
      });
      
      const data = await response.json();
      
      if (!response.ok) {
        throw new Error(data.message || 'Command failed');
      }
      
      // Mark step as completed
      setCompletedSteps((prev) => [...prev, step.id]);

      // If this is the final step, capture the app URL
      if (step.id === steps.length) {
        setAppUrl(data.url || '');
      }
    } catch (err) {
      setError(err.message);
    } finally {
      setLoadingStep(null);
    }
  };

          />

import { render, screen, fireEvent, waitFor } from '@testing-library/react';
import TutorialViewer from './TutorialViewer';

jest.mock('../services/api');

describe('TutorialViewer', () => {
  const mockSteps = [

    { id: 3, title: 'Configure Deployment', description: 'Set up deployment configuration', command: 'az webapp create' },
    { id: 4, title: 'Deploy Code', description: 'Upload application code', command: 'git push' },
    { id: 5, title: 'Verify Deployment', description: 'Check application status', command: 'az webapp show' }
  ];

  beforeEach(() => {
    runCommand.mockResolvedValue({ success: true, output: 'Command executed successfully' });
  });

  test('renders 5 tutorial steps correctly', async () => {
    render(<TutorialViewer />);
    
    expect(screen.getByText('Setup Environment')).toBeInTheDocument();
    expect(screen.getByText('Create App')).toBeInTheDocument();
    expect(screen.getByText('Configure Deployment')).toBeInTheDocument();
    expect(screen.getByText('Deploy Code')).toBeInTheDocument();
    expect(screen.getByText('Verify Deployment')).toBeInTheDocument();
  });

  test('displays progress bar showing 0% initially', () => {
    render(<TutorialViewer />);
    
    const progressBar = screen.getByRole('progressbar');
    expect(progressBar).toHaveStyle('width: 0%');
  });

  test('executes command when Run button is clicked', async () => {
    render(<TutorialViewer />);
    
    const runButton = screen.getByText('Run').closest('button');
    fireEvent.click(runButton);
    
    await waitFor(() => {
      expect(runCommand).toHaveBeenCalledWith('npm install');
    });
  });

  test('updates progress bar when step completes', async () => {
    render(<TutorialViewer />);
    
    const runButton = screen.getByText('Run').closest('button');
    fireEvent.click(runButton);
    
    await waitFor(() => {
      const progressBar = screen.getByRole('progressbar');
      expect(progressBar).toHaveStyle('width: 20%'); // 1/5 steps completed
    });
  });

  test('shows error message when command fails', async () => {
    runCommand.mockRejectedValue(new Error('Command failed'));
    
    render(<TutorialViewer />);
    
    const runButton = screen.getByText('Run').closest('button');
    fireEvent.click(runButton);
    
    await waitFor(() => {
      expect(screen.getByText('Error: Command failed')).toBeInTheDocument();
    });
  });

  test('displays app URL after final step completion', async () => {
    runCommand.mockResolvedValue({
      success: true,
      output: 'https://myapp.azurewebsites.net'
    });
    
    render(<TutorialViewer />);
    
    // Complete all steps
    const runButtons = screen.getAllByText('Run');
    for (let i = 0; i < runButtons.length; i++) {
      fireEvent.click(runButtons[i]);
      await waitFor(() => {
        expect(runCommand).toHaveBeenCalledTimes(i + 1);
      });
    }
    
    expect(screen.getByText('Live Preview')).toBeInTheDocument();
    expect(screen.getByText('https://myapp.azurewebsites.net')).toBeInTheDocument();
